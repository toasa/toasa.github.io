  <form method="GET">
    <input type="text" name="q" placeholder="何か入力">
    <button type="submit">送信</button>
  </form>

  <hr>

  <p>あなたの入力：</p>
  <div id="result">
    <script>
      // URLパラメータを取得
      const params = new URLSearchParams(location.search);
      const q = params.get("q");

      // ⚠ 脆弱：エスケープせずにHTMLとして出力
      if (q) {
        document.write(q);
      }
    </script>
  </div>
</body>
</html>
