ChatGPT に静的ディスパッチと動的ディスパッチのサンプルコードを書いてもらう：

trait Animal {
    fn sound(&self);
}

struct Dog;
struct Cat;

impl Animal for Dog {
    fn sound(&self) {
        println!("Woof!");
    }
}

impl Animal for Cat {
    fn sound(&self) {
        println!("Meow!");
    }
}

// Static dispatch
fn sound(animal: impl Animal) {
    animal.sound();
}

// // Dynamic dispatch
// fn sound(animal: &dyn Animal) {
//     animal.sound();
// }

fn main() {
    let dog = Dog;
    let cat = Cat;

    sound(dog);
    sound(cat);
}
